# Install necessary version of Python & its virtual environment tool, e.g.
sudo apt install python3.7 python3.7.venv


# Python 2:
$ virtualenv venv


# Python 3
$ python3 -m venv venv

# start virtual environment
source env/bin/activate


# exit
deactivate
